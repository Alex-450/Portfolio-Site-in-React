import ArticleLayout from '../Components/ArticleLayout';
import { findBlogPost } from '../utils/blog';

export const blog = findBlogPost(
  (b) => b.type === 'book' && b.topic.toLowerCase() === "the hitchhiker's guide to the galaxy"
);
export const metadata = {
  ...blog,
  bookAuthor: blog.type === 'book' ? blog.author : '',
  keywords: "The Hitchhiker's Guide to the galaxy favourite quotes",
};

"Time is an illusion. Lunchtime doubly so." - p.25

export default function Layout({ children }) {
  return <ArticleLayout metadata={metadata}>{children}</ArticleLayout>;
